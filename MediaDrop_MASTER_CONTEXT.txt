## 🐍 PYTHON BACKEND (downloader.py)

```python
import sys, subprocess, json, os

BIN_PATH = os.path.join(os.path.dirname(__file__), 'bin')
YTDLP = os.path.join(BIN_PATH, 'yt-dlp')
FFMPEG = BIN_PATH

def fetch_info(url):
    result = subprocess.run(
        [YTDLP, '--dump-json', '--no-playlist', url],
        capture_output=True, text=True
    )
    return json.loads(result.stdout)

def download(url, fmt, output_path, subtitle=None):
    cmd = [
        YTDLP, '-f', fmt,
        '--merge-output-format', 'mp4',
        '--ffmpeg-location', FFMPEG,
        '--concurrent-fragments', '8',
        '--continue', '--newline',
        '-o', os.path.join(output_path, '%(title)s.%(ext)s'),
        url
    ]
    if subtitle and subtitle != 'none':
        cmd += ['--write-sub', '--sub-lang', subtitle, '--convert-subs', 'srt']
    process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.STDOUT, text=True)
    for line in process.stdout:
        print(line, flush=True)
    process.wait()
    return process.returncode

def auto_update():
    subprocess.run([YTDLP, '-U'], capture_output=True)

if __name__ == '__main__':
    action = sys.argv[1]
    if action == 'info':
        print(json.dumps(fetch_info(sys.argv[2])))
    elif action == 'download':
        download(sys.argv[2], sys.argv[3], sys.argv[4],
                 sys.argv[5] if len(sys.argv) > 5 else None)
    elif action == 'update':
        auto_update()
```

Progress line format to parse:
```
[download]  45.3% of 120.50MiB at 3.20MiB/s ETA 00:25
```
Extract: %, size, speed, ETA → update UI real-time.
